const SHEET_NAME="Leads"; function doGet(e){const a=e.parameter.action;if(a==="getLeads")return getLeads();return r({error:"unknown"})} function doPost(e){const d=JSON.parse(e.postData.contents);if(d.action==="saveLeads")return saveLeads(d.leads);return r({error:"unknown"})} function getLeads(){const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);const rows=s.getDataRange().getValues();const h=rows[0];const leads=rows.slice(1).filter(row=>row[0]).map(row=>{const o={};h.forEach((k,i)=>o[k]=row[i]);return o});return r({leads})} function saveLeads(leads){const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);const h=s.getRange(1,1,1,s.getLastColumn()).getValues()[0];s.getRange(2,1,Math.max(s.getLastRow()-1,1),h.length).clearContent();if(leads.length>0){const rows=leads.map(l=>h.map(k=>l[k]||""));s.getRange(2,1,rows.length,h.length).setValues(rows)}return r({success:true,count:leads.length})} function r(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON)}
